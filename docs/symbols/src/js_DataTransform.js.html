  7 /**
  8  * @namespace Пакет визуализатора. Содержит основные классы для работы с даными
  9  * (получение, преобразование, отображение)
 10  * @name Visualize
 11  */
 12 
 13 Visualize.DataTransform = atom.Class(
 14     /**
 15      @lends Visualize.DataTransform#
 16      */
 17     {
 18         /**
 19          * @class Класс, отвечающий за преобразование полученных от сервера
 20          * данных в приемлимый для визуализатора формат.
 21          *
 22          * Конструктор класса
 23          * @constructs
 24          * @param {Number} sweep Развертка осциллографа
 25          * @param {Number} countOfPoints Количество точек (пикселей) на канвасе
 26          */
 27         initialize:function (sweep, countOfPoints) {
 28             /**
 29              * Контексn Развертка осцилографа
 30              * @field sweep
 31              * @type Number
 32              */
 33             this.sweep = sweep;
 34 
 35             /**
 36              * Количество точек (пикселей) на канвасе
 37              * @field countOfPoints
 38              * @type Number
 39              */
 40             this.countOfPoints = countOfPoints;
 41 
 42             /**
 43              * Значение, которое характеризует выход за границы
 44              * развертки. Такое же значение установлено в классе
 45              * преобразователя данных
 46              * @field magicOutOfSweep
 47              * @type Number
 48              */
 49             this.magicOutOfSweep = 32568;
 50         },
 51 
 52         /**
 53          * Возвращает преобразованный массив данных.
 54          *
 55          * @method
 56          * @param {Array} array Массив данных для преобразования.
 57          * @param {Function} callback функция, которая будет вызвана после преобразования массива.
 58          */
 59         getFormattedData:function (array, callback) {
 60             var i, j = 0,
 61                 step = 1,
 62                 length = array.length,
 63                 over = 0,
 64                 newArray = [];
 65 
 66             for (i = 0; i < length; i++) {
 67                 if (array[i] > this.sweep || array[i] < -(this.sweep)) {
 68                     array[i] = this.magicOutOfSweep;
 69                 }
 70             }
 71 
 72             if (length > this.countOfPoints) {
 73                 for (i = 2; i <= length / this.countOfPoints; i += 2) {
 74                     over = length / i;
 75                     if (over === this.countOfPoints) {
 76                         step = i;
 77                         break;
 78                     } else if (over < this.countOfPoints) {
 79                         for (j = 0; j < length - 1; j++) {
 80                             newArray[j] = array[j];
 81                             newArray[j + 1] = (array[j] + array[j + 1]) / 2;
 82                             callback(newArray);
 83                         }
 84                         break;
 85                     }
 86                 }
 87             } else if (length < this.countOfPoints) {
 88                 for (j = 0; j < this.countOfPoints; j += 2) {
 89                     if (j + 1 <= length) {
 90                         newArray[j] = array[j / 2];
 91                         newArray[j + 1] = (array[j / 2] + array[j / 2 + 1]) / 2;
 92                     } else {
 93                         newArray[j] = array[j / 2];
 94                         newArray[j + 1] = (array[j / 2] + array[j / 2] + 1) / 2;
 95                     }
 96                 }
 97                 callback(newArray);
 98             }
 99 
100             for (i = 0; i < length / step; i += 1) {
101                 newArray[i] = array[i * step];
102             }
103 
104             callback(newArray);
105         },
106 
107         /**
108          * Изменяет значение развертки.
109          *
110          * @method
111          * @param newSweep Новое значение развертки.
112          */
113         changeSweep:function (newSweep) {
114             this.sweep = newSweep;
115         }
116     });
